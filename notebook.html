Growth Calculations

Compute Compound Annual Growth Rate (CAGR) for ridership at each station over different time periods (2001-2019 and 2021-2024) to identify top gainers and losers.

In[131]:
# Calculate CAGR for 2021-2024 period
# Filter data for years 2021 to 2024
data_21_24 = yearly_avg_by_station[(yearly_avg_by_station['date'] >= 2021) & (yearly_avg_by_station['date'] <= 2024)]

# Function to compute CAGR: (final/initial)^(1/n) - 1, where n is number of years
def compute_cagr_21_24(group):
    if len(group) < 2:
        return pd.Series({'cagr': np.nan})
    initial_series = group[group['date'] == 2021]['overall_avg']
    final_series = group[group['date'] == 2024]['overall_avg']
    if initial_series.empty or final_series.empty:
        return pd.Series({'cagr': np.nan})
    initial = initial_series.values[0]
    final = final_series.values[0]
    n = 2024 - 2021  # 3 years
    if initial > 0:
        cagr = (final / initial)**(1/n) - 1
    else:
        cagr = np.nan
    return pd.Series({'cagr': cagr})

# Apply CAGR calculation to each station group
growth_df_21_24 = data_21_24.groupby('stationname').apply(compute_cagr_21_24).reset_index()

# Convert CAGR to percentage
growth_df_21_24['growth_percentage'] = growth_df_21_24['cagr'] * 100
growth_df_21_24 = growth_df_21_24.drop(columns=['cagr'])

# Round to 2 decimal places and sort descending
growth_df_21_24['growth_percentage'] = growth_df_21_24['growth_percentage'].round(2)
growth_df_21_24 = growth_df_21_24.dropna(subset=['growth_percentage'])
growth_df_21_24 = growth_df_21_24.sort_values(by='growth_percentage', ascending=False)

# Custom icon function for cluster averaging colors
icon_create_function = r"""
function(cluster) {
    // Helper functions to convert between hex and RGB
    function hexToRgb(hex) {
        let result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
        return result ? {
            r: parseInt(result[1], 16),
            g: parseInt(result[2], 16),
            b: parseInt(result[3], 16)
        } : null;
    }
    function componentToHex(c) {
        let hex = c.toString(16);
        return hex.length == 1 ? "0" + hex : hex;
    }
    function rgbToHex(r, g, b) {
        return "#" + componentToHex(r) + componentToHex(g) + componentToHex(b);
    }

    let childMarkers = cluster.getAllChildMarkers();
    let total_r = 0, total_g = 0, total_b = 0;
    
    // Sum the RGB values of all markers in the cluster
    childMarkers.forEach(function(marker) {
        let markerColor = marker.options.fillColor;
        let rgb = hexToRgb(markerColor);
        if (rgb) {
            total_r += rgb.r;
            total_g += rgb.g;
            total_b += rgb.b;
        }
    });
    
    // Calculate the average RGB
    let count = childMarkers.length;
    let avg_r = Math.round(total_r / count);
    let avg_g = Math.round(total_g / count);
    let avg_b = Math.round(total_b / count);
    
    let avg_hex_color = rgbToHex(avg_r, avg_g, avg_b);
    
    let childCount = cluster.getChildCount();
    let c = ' marker-cluster-';
    if (childCount < 10) {
        c += 'small';
    } else if (childCount < 100) {
        c += 'medium';
    } else {
        c += 'large';
    }

    // Create and return the custom icon
    return new L.DivIcon({ 
        html: `<div style="background-color: ${avg_hex_color};"><span>${childCount}</span></div>`, 
        className: 'marker-cluster' + c, 
        iconSize: new L.Point(40, 40) 
    });
}
"""

# Calculate CAGR for each line from 2001 to each subsequent year up to 2019
growth_df = yearly_line_sum.copy()
for line in growth_df.index:
    growth_df.loc[line, 2001] = np.nan  # Set base year to NaN
    for year in range(2002, 2020):
        start_val = yearly_line_sum.loc[line, 2001]
        end_val = yearly_line_sum.loc[line, year]
        periods = year - 2001
        cagr = (end_val / start_val) ** (1 / periods) - 1
        growth_df.loc[line, year] = round(cagr * 100, 2)

# Extract CAGR data for 2002-2019
cagr_2001_2019 = growth_df.loc[:, 2002:2019]

# Calculate average CAGR for each line over the period
average_cagr = cagr_2001_2019.mean(axis=1).sort_values(ascending=False)
overall_mean_cagr = average_cagr.mean()

plt.title('Average CAGR from 2001 to 2019 by CTA Line')
plt.ylabel('Average CAGR (%)')
